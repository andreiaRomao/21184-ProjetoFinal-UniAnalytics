\core\analytics\indicator\any_access_after_end

SELECT COUNT(*)
FROM mdl_logstore_standard_log
WHERE userid = :userid
  AND contextlevel = :contextlevel
  AND contextinstanceid = :contextinstanceid
  AND timecreated > :start;
  
(com as tabelas)
SELECT timeaccess
FROM mdl_user_lastaccess
WHERE userid = :userid
  AND courseid = :courseid;

\core\analytics\indicator\any_course_access

SELECT COUNT(*)
FROM mdl_logstore_standard_log
WHERE userid = :userid
  AND courseid = :courseid
  [AND timecreated > :starttime]
  [AND timecreated <= :endtime];

(com as tabelas)
SELECT userid, timeaccess
FROM mdl_user_lastaccess
WHERE courseid = :courseid;

FROM mdl_logstore_standard_log

    contextinstanceid = <id_instancia> AND
    (crud = 'c' OR crud = 'u')
