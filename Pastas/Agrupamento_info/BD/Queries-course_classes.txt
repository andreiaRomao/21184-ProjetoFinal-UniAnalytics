\core_course\analytics\indicator\no_student
 
 (com as tabelas)
SELECT u.id
FROM mdl_user u
JOIN mdl_role_assignments ra ON ra.userid = u.id
JOIN mdl_context ctx ON ctx.id = ra.contextid
WHERE ctx.id = :contextid
  AND ra.roleid = :roleid;
